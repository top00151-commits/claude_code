/* OPS-W-3 [P2]: 주간 보고 인쇄 룰 (사이드바·도크·헤더 숨김) */

  .ws-tabs, nav.ws-tabs,
  button:not(.print-keep), .btn, .btn-sm {
    display: none !important;
  }
  .main, main.main, .content {
    margin: 0 !important;
    padding: 0 !important;
    width: 100% !important;
    max-width: none !important;
  }
  body { background: #fff !important; color: #000 !important; }
  .weekly-card, .weekly-grid, .weekly-table {
    page-break-inside: avoid;
  }
  /* CEO 직접 인쇄 시 빅터 인사이트 카드 print 표시 보장 */
  .weekly-insight, .victor-insight {

    page-break-before: always;
  }
}
</style>
