## 5. JS 정식 사양 (base.html 의 `<script>` 블록)

```javascript
/* 옛 ws-wrap dropdown JS 폐기 — 가로 탭은 단순 <a> 링크 */
/* (기존 wsBtn, wsSet, wsMenu 관련 JS 모두 제거) */

/* 빅터 도크 모드 탭 전환 */
function setDockMode(mode){
  document.querySelectorAll('.dock-mode-tab').forEach(t => {
    const a = t.dataset.mode === mode;
    t.classList.toggle('active', a);
    t.setAttribute('aria-selected', a ? 'true' : 'false');
  });
  document.querySelectorAll('.dock-mode-panel').forEach(p => {
    p.classList.toggle('active', p.dataset.modePanel === mode);
  });
  if (mode === 'search') {
    setTimeout(() => {
      const i = document.getElementById('gSearchInput');
      if (i) { i.focus(); i.select(); }
    }, 100);
  }
}
document.querySelectorAll('.dock-mode-tab').forEach(t => {
  t.addEventListener('click', () => setDockMode(t.dataset.mode));
});

/* "/" 단축키 → 도크 열기 + 검색 모드 + 검색 input 포커스 */
document.addEventListener('keydown', function(e){
  if (e.key !== '/' || e.ctrlKey || e.metaKey || e.altKey || e.isComposing) return;
  const t = (e.target.tagName || '').toLowerCase();
  if (t === 'input' || t === 'textarea' || t === 'select' || e.target.isContentEditable) return;
  e.preventDefault();
  /* 도크 닫혀있으면 열고 */
  if (!document.body.classList.contains('dock-open') && typeof victorToggleDock === 'function') {
    victorToggleDock();
  }
  /* 검색 모드 활성화 (자동 포커스 포함) */
  setTimeout(() => setDockMode('search'), 280);
});
```
