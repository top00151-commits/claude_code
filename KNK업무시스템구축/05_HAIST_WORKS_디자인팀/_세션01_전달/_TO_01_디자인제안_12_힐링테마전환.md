#### (E) toggleDock() 표준 구현

```js
/* 상태 유지: localStorage */
const DOCK_KEY = 'haist_victor_dock_open';

function setDockOpen(open, persist = true) {
  document.body.classList.toggle('dock-collapsed', !open);
  const btn = document.querySelector('.tb-icon-btn.victor');
  if (btn) btn.setAttribute('aria-pressed', open ? 'true' : 'false');
  if (persist) localStorage.setItem(DOCK_KEY, open ? '1' : '0');
}

function toggleDock() {
  const collapsed = document.body.classList.contains('dock-collapsed');
  setDockOpen(collapsed);   /* collapsed 였다면 열기 */
}

/* 초기 상태 복원 */
document.addEventListener('DOMContentLoaded', () => {
  const saved = localStorage.getItem(DOCK_KEY);
  const open = saved === null ? true : saved === '1';  /* 기본: 열림 */
  setDockOpen(open, false);
});

/* Ctrl+K 단축키 */
document.addEventListener('keydown', (e) => {
  if ((e.ctrlKey || e.metaKey) && e.key.toLowerCase() === 'k') {
    e.preventDefault();
    toggleDock();
  }
  /* Esc: 도크가 열려 있고 textarea 포커스 중이면 접기 */
  if (e.key === 'Escape' && !document.body.classList.contains('dock-collapsed')) {
    const active = document.activeElement;
    if (active && active.closest('.dock')) setDockOpen(false);
  }
});
```
